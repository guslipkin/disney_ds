for (i in 1:14) {
  chart <- rides[[i]] %>%
    mutate(weekday = weekdays(datetime)) %>%
    group_by(weekday) %>%
    summarise(mean_wait = mean(SPOSTMIN, na.rm = TRUE)) %>%
    ggplot(aes(x = ordered(weekday, levels = days_of_week), y = mean_wait)) +
    geom_col() +
    labs(title = names(rides[i]))
  print(chart)
}
